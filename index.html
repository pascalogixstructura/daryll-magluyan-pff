This is the code logic used by the dropdown clock. It reads the selected time zone and formats the date/time accordingly.

const picker = document.getElementById('timezonePicker');
const refreshBtn = document.getElementById('refreshClock');

function formatTimeForZone(timeZone) {
  return new Intl.DateTimeFormat('en-US', {
    timeZone,
    hour: 'numeric',
    minute: '2-digit',
    second: '2-digit',
    hour12: true
  }).format(new Date());
}

function formatDateForZone(timeZone) {
  return new Intl.DateTimeFormat('en-US', {
    timeZone,
    weekday: 'long',
    year: 'numeric',
    month: 'long',
    day: 'numeric'
  }).format(new Date());
}

function updateClocks() {
  const selectedZone = picker.value;
  const selectedLabel = picker.options[picker.selectedIndex].dataset.label;

  document.getElementById('selectedTime').textContent = formatTimeForZone(selectedZone);
  document.getElementById('selectedDate').textContent = formatDateForZone(selectedZone);
  document.getElementById('selectedZone').textContent = selectedLabel;

  document.getElementById('phtTime').textContent = formatTimeForZone('Asia/Manila');
  document.getElementById('phtDate').textContent = formatDateForZone('Asia/Manila');
}

picker.addEventListener('change', updateClocks);
refreshBtn.addEventListener('click', updateClocks);

// Initialize on page load
updateClocks();
